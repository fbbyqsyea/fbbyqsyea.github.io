常用Docker

Sun, Oct 23, 2022

mysql

$ docker run --name mysql \
-p 3306:3306 \
--privileged=true \
--restart=always \
-e MYSQL_ROOT_PASSWORD='Aa!23456' \
-d mysql:5.7

redis

$ docker run --name redis \
-p 6379:6379 \
--privileged=true \
--restart=always \
-d redis redis-server

etcd

$ docker run -d --name Etcd-server \
    -p 2379:2379 \
    -p 2380:2380 \
    --privileged=true \
    --restart=always \
    --env ALLOW_NONE_AUTHENTICATION=yes \
    bitnami/etcd:latest
